# eslint 如何根据文件不同使用不通过规则

  1. 升级 eslint 到最新版，并npm install --save-dev typescript
  2. 在.eslintrc.js 中设置
"overrides": [
    {
      "files": ['*.js'],
      "parser": 'babel-eslint',
    },
    {
      "files": ['*.ts', '*.tsx'],
      "extends": ["plugin:@typescript-eslint/recommended"],
      "parser": '@typescript-eslint/parser',
      "plugins": ['@typescript-eslint/eslint-plugin'],
    },
  ]
